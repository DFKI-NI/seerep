280template <typename T>
281class PointCloud2Iterator
282 : public impl::PointCloud2IteratorBase<T, T, unsigned char, seerep::PointCloud2, PointCloud2Iterator>
283{
284public:
291 PointCloud2Iterator(seerep::PointCloud2& cloud_msg, const std::string& field_name)
292 : impl::PointCloud2IteratorBase<T, T, unsigned char, seerep::PointCloud2,
293 seerep_hdf5_pb::PointCloud2Iterator>::PointCloud2IteratorBase(cloud_msg, field_name)
294 {
295 int offset = this->set_field(cloud_msg, field_name);
296 this->data_char_ = reinterpret_cast<unsigned char*>(cloud_msg.mutable_data()->data() + offset);
297 this->data_ = reinterpret_cast<T*>(this->data_char_);
298 this->data_end_ = reinterpret_cast<T*>(cloud_msg.mutable_data()->data() + cloud_msg.data().size() + offset);
299 }
300};
