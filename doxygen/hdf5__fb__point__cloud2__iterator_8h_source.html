120template <typename T>
121class PointCloud2Iterator
122 : public impl::PointCloud2IteratorBase<T, T, unsigned char, seerep::fb::PointCloud2, PointCloud2Iterator>
123{
124public:
134 PointCloud2Iterator(uint8_t* data, uint32_t offset, uint32_t pointStep, uint32_t height, uint32_t width)
135 : impl::PointCloud2IteratorBase<T, T, unsigned char, seerep::fb::PointCloud2,
136 seerep_hdf5_fb::PointCloud2Iterator>::PointCloud2IteratorBase()
137 {
138 this->field_offset_ = offset;
139 this->point_step_ = pointStep;
140 this->data_char_ = reinterpret_cast<unsigned char*>(data + this->field_offset_);
141 this->data_ = reinterpret_cast<T*>(this->data_char_);
142 // last element is at: offset + (n -1) * pointStep
143 this->data_end_ = reinterpret_cast<T*>(data + offset + (height * width - 1) * pointStep);
144 }
145};
